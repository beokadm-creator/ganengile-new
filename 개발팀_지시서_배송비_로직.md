# 개발팀 지시서: 1단계(지하철 to 지하철) 배송비 로직 수정

## 📋 작업 개요

**작업명:** 1단계 배송비 로직 수정 (Phase 1 Delivery Fee Logic Update)
**우선순위:** 🔴 긴급 (Highest)
**목표일:** 2026년 3월 16일 (1주일 내 완료)
**담당자:** 프론트엔드/백엔드 팀 전체

---

## 🎯 작업 목표

1. **기존 배송비 로직을 1단계(지하철 to 지하철) 최적화 로직으로 교체**
2. **길러 정산 시스템 구현** (길러 85%, 플랫폼 15%)
3. **지하철 역 개수 기반 동적 거리 계산 구현**
4. **테스트 코드 작성 및 검증**

---

## 📊 변경사항 요약

### 핵심 7가지 변경

| 항목 | 변경 전 | 변경 후 | 영향도 |
|------|--------|--------|--------|
| **기본료** | 3,000원 | 3,500원 | 🔴 높음 |
| **거리료** | 800원 (고정) | 600~2,400원 (역 개수 기반) | 🔴 높음 |
| **서비스 수수료** | 0원 | 15% | 🔴 높음 |
| **길러 정산** | 없음 | 85% | 🔴 높음 |
| **최소 배송비** | 없음 | 3,000원 | 🟡 중간 |
| **최대 배송비** | 없음 | 8,000원 | 🟡 중간 |
| **긴급 surcharge** | baseFee × % | (base+distance) × % | 🟡 중간 |

#### 2.1 새로운 배송비 계산 함수 구현

**파일:** `src/utils/deliveryFee.js`

```javascript
/**
 * 배송비 계산 함수 (1단계: 지하철 to 지하철)
 * @param {Object} params - 배송 매개변수
 * @param {number} params.stationCount - 지하철 역 개수 (필수)
 * @param {number} params.weight - 무게 (kg)
 * @param {string} params.packageSize - 패키지 사이즈 (small, medium, large, xlarge)
 * @param {string} params.urgency - 긴급도 (normal, fast, urgent)
 * @returns {Object} 계산된 배송비 정보
 */
function calculatePhase1DeliveryFee(params) {
  const {
    stationCount = 5,
    weight = 1,
    packageSize = 'small',
    urgency = 'normal'
  } = params;

  // 기본 설정
  const BASE_FEE = 3500;
  const MIN_FEE = 3000;
  const MAX_FEE = 8000;
  const PLATFORM_FEE_RATE = 0.15;

  // 1. 기본 배송비
  const baseFee = BASE_FEE;

  // 2. 거리 수수료 (역 개수 기반)
  const distanceFee = calculatePhase1DistanceFee(stationCount);

  // 3. 무게별 추가 요금
  const weightFee = calculateWeightFee(weight);

  // 4. 사이즈별 추가 요금
  const sizeFee = calculateSizeFee(packageSize);

  // 5. 긴급도 surcharge
  const urgencySurcharge = calculateUrgencySurcharge(urgency, baseFee + distanceFee);

  // 6. 서비스 수수료
  const serviceFee = calculateServiceFee(baseFee + distanceFee + weightFee + sizeFee, PLATFORM_FEE_RATE);

  // 7. 부가세 (VAT 10%)
  const subtotal = baseFee + distanceFee + weightFee + sizeFee + urgencySurcharge + serviceFee;
  const vat = Math.round(subtotal * 0.1);

  // 8. 최종 배송비
  let totalFee = subtotal + vat;

  // 최소/최대 제한
  if (totalFee < MIN_FEE) totalFee = MIN_FEE;
  if (totalFee > MAX_FEE) totalFee = MAX_FEE;

  // 9. 길러 비용 분배
  const breakdown = calculatePhase1Breakdown(totalFee, urgency);

  return {
    baseFee,
    distanceFee,
    weightFee,
    sizeFee,
    urgencySurcharge,
    serviceFee,
    subtotal,
    vat,
    totalFee,
    breakdown,
    description: `1단계 배송 (역 ${stationCount}개구간, ${packageSize}, ${urgency})`
  };
}

/**
 * 거리 수수료 계산 (1단계: 지하철 역 개수 기반)
 */
function calculatePhase1DistanceFee(stationCount) {
  const baseStations = 5;
  const feePerStation = 120;

  if (stationCount <= baseStations) {
    return 600;
  }

  return 600 + (stationCount - baseStations) * feePerStation;
}

/**
 * 무게별 추가 요금
 */
function calculateWeightFee(weight) {
  const baseWeight = 1;
  const feePerKg = 100;

  if (weight <= baseWeight) return 100;
  return Math.round(weight * feePerKg);
}

/**
 * 사이즈별 추가 요금
 */
function calculateSizeFee(packageSize) {
  const sizeFees = {
    small: 0,
    medium: 400,
    large: 800,
    xlarge: 1500
  };

  return sizeFees[packageSize] || 0;
}

/**
 * 긴급도 surcharge
 */
function calculateUrgencySurcharge(urgency, baseAndDistanceFee) {
  const urgencyMultipliers = {
    normal: 0,
    fast: 0.1,
    urgent: 0.2
  };

  const multiplier = urgencyMultipliers[urgency] || 0;
  return Math.round(baseAndDistanceFee * multiplier);
}

/**
 * 서비스 수수료
 */
function calculateServiceFee(feeBeforeService, rate = 0.15) {
  return Math.round(feeBeforeService * rate);
}

/**
 * 길러 비용 분배 (1단계: 길러 1명)
 */
function calculatePhase1Breakdown(totalFee, urgency) {
  const platformFee = Math.round(totalFee * 0.15);
  const remainingFee = totalFee - platformFee;
  const gillerFee = remainingFee;

  return {
    platformFee,
    gillerFee,
    total: totalFee
  };
}

module.exports = {
  calculatePhase1DeliveryFee,
  calculatePhase1DistanceFee,
  calculateWeightFee,
  calculateSizeFee,
  calculateUrgencySurcharge,
  calculateServiceFee,
  calculatePhase1Breakdown
};
```

describe('1단계 배송비 계산', () => {

      weight: 1,
      packageSize: 'small',

    const normal = calculatePhase1DeliveryFee({ stationCount: 5, urgency: 'normal' });
    const fast = calculatePhase1DeliveryFee({ stationCount: 5, urgency: 'fast' });
    const urgent = calculatePhase1DeliveryFee({ stationCount: 5, urgency: 'urgent' });

    expect(fast.totalFee).toBeGreaterThan(normal.totalFee);
    expect(urgent.totalFee).toBeGreaterThan(fast.totalFee);
  });
